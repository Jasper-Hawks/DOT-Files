*** GENERAL Keybinds
#+begin_src emacs-lisp
  ;; general package
  (use-package general
  :config
    (general-evil-setup)
    (general-create-definer Jasper/leader
    :states '(general normal insert visual emacs)
    :keymaps 'override
    :prefix "SPC"
    :non-normal-prefix "M-SPC")


    (Jasper/leader
    "!" 'shell-command
    ":" 'eval-expression)

    ;; Buffer prefix
    (Jasper/leader
    "b" '(:ignore t :wk "buffer")
    "bd" '(kill-buffer :wk "Kill buffer")
    "<" '(switch-to-buffer :wk "Switch buffer")
    )

    ;; Window prefix
    (Jasper/leader
    "w" '(:ignore t :wk "window")
    "wr" '(split-window-right :wk "Split window right")
    "wd" '(kill-buffer :wk "Kill window")
    )
    ;File prefix
    (Jasper/leader
    "f" '(:ignore t :wk "file")
    "ff" '(find-file :wk "Find file")
    "fF" '(sudo-edit :wk "Sudo find file")
    )
  )
#+end_src
